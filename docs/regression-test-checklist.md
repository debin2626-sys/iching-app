## 快速执行命令

```bash
# 本地构建验证
cd /root/.openclaw/workspace/developer/iching-app
npm run build 2>&1 | tail -20

# 启动生产模式测试
npm start &
sleep 3

# 快速 API 健康检查
curl -s http://localhost:3000/api/health | python3 -m json.tool

# 检查各语言路由
for locale in zh zh-TW en; do
  status=$(curl -s -o /dev/null -w "%{http_code}" http://localhost:3000/$locale)
  echo "$locale: $status"
done

# 检查安全响应头
curl -sI http://localhost:3000 | grep -E "Strict-Transport|X-Frame|Cache-Control"
```
